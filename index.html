{

  let count=0;
  for(let j=0;j < n;j++){
    if(str[i]===str[j] && i>j){
      break;
    }
    if(str[i]===str[j]){
      count+=1;
    }
  }
  if(count>1){
    ans+=str[i]+count;
  }
}
return ans;
};

var Reverse_String = (str) => 
{
  let result="";
  for(let i=str.length-1;i >= 0;i--){
    result+=str.charAt(i);
  }
  return result;
};
